create TABLE Forum(
    [Id] int primary key IDENTITY(1,1),
    [Title] NVARCHAR(max),
    [Description] NVARCHAR(max),
    [Like] int,
    [Dislike] int,
)

CREATE TABLE [Log](
    log_id INT PRIMARY KEY IDENTITY(1,1),
    userid INT,
    url NVARCHAR(MAX),
    method_type NVARCHAR(MAX),
    status_code INT,
    request_body NVARCHAR(MAX),
    response_body NVARCHAR(MAX)
);
