create TABLE Forum(
    [Id] int primary key IDENTITY(1,1),
    [Title] NVARCHAR(max),
    [Description] NVARCHAR(max),
    [Like] int,
    [Dislike] int,
)


create table Log(
    [log-id] int primary key IDENTITY(1,1),
    [userid] NVARCHAR(max),
    [url] NVARCHAR(max),
    [method-type] NVARCHAR(max),
    [status-code] int,
    [request body] NVARCHAR(max),
    [response body] NVARCHAR(max),
)
